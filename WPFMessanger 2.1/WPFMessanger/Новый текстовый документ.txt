CREATE procedure EditData
@ID int,
@FName varchar(35),
@LName varchar(35),
@Country varchar(35),
@age int,
@Description Text,
@Photo Text
as begin
Update Account
Set FirsName=@FName,
 LastName=@LName,
 country=@Country,
 Age=@age,
 Description=@Description,
 PathOfPhoto=@Photo
where(id=@ID);
end

Create function Auentific(@Login varchar(50),@Password  varchar(50))
returns varchar(50)
as
BEGIN
Declare @ret varchar(50), @tmp varchar(50);
set @ret=0
--set @tmp=''
	Select @ret=Account.FirsName
	From Account
	group by Login,password,FirsName
	Having Login=@Login AND Password=@Password
	--	if(@tmp!=0)
	--	set @ret=Account.ID
	return @ret;
END

create procedure LoadImage
@Photo TEXT,
@id int
as 
begin 
UPDATE Account  
SET PathOfPhoto = @Photo  
WHERE id = @id;  
end


create procedure SendMessage
@Mess TEXT,
@id int,
@To int
as 
begin 
UPDATE Messag  
SET content = @Mess  
WHERE (idAccount = @id and ToAcc=@To) or(idAccount =@To  and ToAcc=@id);  
end
