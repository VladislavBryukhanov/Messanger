CREATE procedure EditData
@ID int,
@FName varchar(35),
@LName varchar(35),
@Country varchar(35),
@age int,
@Description Text,
@Photo Text
as begin
Update Account
Set FirsName=@FName,
 LastName=@LName,
 country=@Country,
 Age=@age,
 Description=@Description,
 PathOfPhoto=@Photo
where(id=@ID);
end 


create procedure AddToCompanion
@isWriteField TEXT,
@id int
as 
begin 
UPDATE Relations  
SET isWrite = @isWriteField  
WHERE (idOfOwner = @id ) ;  
end

create procedure AddToFriends
@isfriendField TEXT,

Create function Auentific(@Login varchar(50),@Password  varchar(50))
returns varchar(50)
as
BEGIN
Declare @ret varchar(50), @tmp varchar(50);
set @ret=0
--set @tmp=''
	Select @ret=Account.FirsName
	From Account
	group by Login,password,FirsName
	Having Login=@Login AND Password=@Password
	--	if(@tmp!=0)
	--	set @ret=Account.ID
	return @ret;
END

create procedure SendMessage
@Mess TEXT,
@id int,
@To int
as 
begin 
UPDATE Messag  
SET content = @Mess
